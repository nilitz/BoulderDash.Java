30  	}

34  	 *

39  	}
40  
41  	/**

43  	 *
44  	 * @param viewFrame
45  	 *          the new view frame
46  	 */

48  		this.viewFrame = viewFrame;
49  	}
50  
51  	/*
52  	 * (non-Javadoc)
53  	 *
54  	 * @see java.util.Observer#update(java.util.Observable, java.lang.Object)
55  	 */
56  	public void update(final Observable arg0, final Object arg1) {
57  		this.repaint();
58  	}
59  
60  	/*
61  	 * (non-Javadoc)
62  	 *
63  	 * @see javax.swing.JComponent#paintComponent(java.awt.Graphics)
64  	 */

68  		graphics.drawString(this.getViewFrame().getModel().getHelloWorld().getMessage(), 10, 20);
69  	}
